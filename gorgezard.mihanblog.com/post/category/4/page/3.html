بزن شلاق به پشتم
                                        منی که تنهایی نوشتم
بگو بخشیدی قلبمو
                                         بگو قیامت قلبمو
نخواستم برسم،ببخش
                                         تنها و بی کسم، ببخش
دیروز مست غرورم
                                          امروز ازت دورم
دیروز خنده ی گناهم
                                         امروز گریه ی آگاهم
عمرم تلخی و کوتاهِ
                                         جاده ی چشمام گمراهِ
نمی خوام بمونم،قیامت
                                          بی وفایم بمونه یادت

دوست دارم، پشیمونم
                                         واست بیقرارم میدونم

همیشه غمگینِ برده
                        سهمی از دنیا نَبُرده
تو بگو عدالت چیه؟
                        زنجیر من دست کیه؟
بی گناهم نزن کنایه
                        روسرم مونده سایه
ابری شده نگاهم 
                        بازم برده و گناهم
آهای خنده ات نگیره
                        آهم دامنتو میگیره
بسته خسته ام دیره
                        دنیام سیاه و تیره
بذار برم کی بردم
                        میرم دیگه بر نمی گردم
نمی خوام سلول و بندم
                        اینجاشو  فریاد بلندم
من خلوت ترم از همیشه
                        خشک و سرده ریشه
طلسم نیست می دونم
                        بذار پر بزنم میتونم
بنویس حکم آزادی
                        توی که محبت ندادی
پیله رو پینه کردم
                        پیراهنُ پاره کردم
یاد خونه ام، بنویس
                       قسم به چشمای خیس

این بار تو بس کن
                       زنجیرمُ مرخص کن
دل خونِ و دلگیر
                       نذار بمونه تقدیر
ابریِ ماه و مهتابم
                       غصه گرفته و بی تابم
الکی نوازشم نکن
                       قبرُ عمیق تر نکن
شاید بارون، شاید
                       تنها می مونم باید
آره اسیر و پا برهنه
                       پنهون نمایش صحنه
انتقام و کینه ندارم
                       می رم از تو گله ندارم
فقط بذاری من برم
                       کارهاتو از یاد ببرم
زردِ درخت وجودم
                       لُختِ رخت وجودم
صبر و سلامتی ندارم
                       ازاسیری خودم بیزارم
قصه ی روزگارمُ گفتم
                       دلیل و معیارمُ گفتنم
حالا یه لطفی کن
                       دست بندُ باز کن
