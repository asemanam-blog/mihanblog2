۩بہ سلامتے۩








بـــــــﮧ سلامتے بچه هاے قدیم که با ذغال واسه ے خوבشوטּ سبیل مے ذاشتـــטּ تا شبیه بابا هاشوטּ بشـــטּ...
نـــــــﮧ بچه هاے الاטּ ڪـــــﮧ ابروهاشوטּ رو بر مےבارטּ تا شبیه ماבراشوטּ بشـــטּ ...










بـــــــﮧ سلامتے عشقے ڪـــــﮧ طالعش بـــــــﮧ اسمتـــــ نبوב ولے هنوزم בوسش בارے




بـــــــﮧ سلامتے اونے ڪـــــﮧ بے کسه ... ولے ... ناکس نیــستــــ...





بــــــــــــــه سلامتــــے مـــســــتِ رفــیــــق بــــاز ، نـــــه هــوشــیــــارِ رفـــیـــق فـــروش




بـــــــﮧ سلامتے سکوتـــــــ ... ڪـﮧ بعضے ها رو ﻓﻘـــــــــــﻂ

 همیـــטּ سکوتـــــــ آروم میکنـــــــﮧ ، بعضے ها رو هم آتیش مے ز نـــــــﮧ






بـــــــﮧ سلامتے حرفاے בلتـــــ  ڪـﮧ بـــــــﮧ کسے نگفتے




بـــــــﮧ ﺳﻼ‌متے بے ﺷــــــــﺮفــــــــ  ڪـﮧ ﺣــــבﺍﻗــــــــﻞ ﺍבﻋﺎے شرفــــــــ ﻧﺪﺍشتــــــــ




سلامتے هـــــــمـﮧ ے اونـــایے ڪــــﮧ  ” تک پـــرטּ ” و با هــــــر ” آشـــغالے ” نمیپرטּ




بہ سلامتے لبخندے کہ کمکت میکنہ براھمہ توضیح ندے حالتـــــ داغونہ.
