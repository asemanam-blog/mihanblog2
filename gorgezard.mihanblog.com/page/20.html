                         تنها بمیرم ، بی علت

از همه دلگیرم ، خدایا

                         سیاه تقدیرم ، خدایا

فریب کاری تا کی

                        گریه و بیقراری تا کی

شبهام نداره ستاره

                       بی درمون قلب بیچاره

بهونه نمی خوام از کسی

                      نمی کشم لحظه ای دلواپسی

دیونگیِ ، تردید ندارم

                      از اینا انتظاری ندارم

عشق اینها از صنعتِ

                       تنهایی واسم یه نعمتِ

اولش دلدادگی و مهربونی ِ

                       آخرش وجودم زندونی ِ

از پشت خنجر می زنن

                       به صورتم سیلی می زنن

خدایا نخواستیم و نمیشه

                     با اینها خشک این ریشه

فهمیدم و چشیدم ، بسته

                     تموم غرورم ، شکسته

نمونده اشکی از چشمام

                     سرد و کبودِ این دستام

گفتن شمع ، پروانه ام

                     گفتن شراب ، میخانه ام

گفتن سکوت ، بی صدام

                      گفتن نفرت ، تنهام

گفتن و صدبار گفتن

                      از مردنم ، تکرار گفتن

نرگس بردم ، پرپره

                      قلب بردم ، در به دره

شقایق بردم ، مُرده

                      قناری بردم ، مُرده

محبت کردم ، نخواستن

                     رفاقت کردم ، نخواستن

خسته ام از عشق و عاشقی

                      اینها رو نمی خوام دقایقی

شدم خسرو ، بسته

                        غصه کنارم نشسته

خدایا گناهی ندارم

                       از زمونه پناهی ندارم

خنده رو یادم نمی دن

                        به من راست نمی گن

من ساده صبر کشیدم

                        مزه ی غصه رو چشیدم

اینها نمی دن جوابی

                         مستم نمی کنن به شرابی

توبه ، صد بار توبه

                          غصه خوردنم خوبه

نمی خوام ، مبهم شدم

                           تکه ای ازجهنم شدم

پشیمونم ، عاشق نمی شم

                           با اینها صادق نمی شم

خیلی خالیه ، خیالم

                           شکست خورده و بی حالم

خیلی غصه هام زیاده

                           لعنت به این دل ساده

راهی، نگاهی می خوام

                           از خودت توجهی می خوام

می نویسم اعتراض و نیاز

                             شاید باشی یه همراز

عشق رو گدایی نمی کنم

                             دستمو دراز نمی کنم

گاهی به خودت می گم

                            زخم ها رو از یاد می برم

خدایا، خدایی از خودم

                             تویی بهونه ی موندنم

می خوام ساقی باشم

                             شقایق و عقاقی باشم

شاد و خوشحال بشم

                             غرق در خیال بشم

دستتو بگیرم همیشه

                            رفیقی من و تو ، میشه

ساده و سلامتم، خدایا

                             از عشقت قناعتم، خدایا

شیرینه لحظه های تلخم

                              آب میشه سرما و یخم

تو به من بده ، امید

                              امروز دلم به تو رسید

تو غفوری ، پر از نوری

                              از دنیای اینها دوری

رهام ، تو صدام کن

                              تو سیاهی پیدام کن

خدایا اجازه می خوام

                             وجودت نیازه ، می خوام

تمومه فصل خزونم

                             بهاره، وصل خزونم

شعر این کتابم تویی

                             واقعیت سرابم تویی

همیشه، روز قرارمون ِ

                             عشق، پایان انتظارمون ِ
