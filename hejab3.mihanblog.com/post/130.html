  • حضرت زهرا (ع) و حجاب:

    امیرالمؤمنین علیه السلام می‌فرماید: روزی رسول خدا از ما پرسید: بهترین کار برای زنان چیست؟

    فاطمه علیهاسلام پاسخ داد:« بهترین کار برای زنان این است که مردان را نبینند و مردان نیز آنها را نبینند .
    یَا مُمْتَحَنَةُ امْتَحَنَكِ اللَّهُ الَّذِی خَلَقَكِ قَبْلَ أَنْ یَخْلُقَكِ فَوَجَدَكِ لِمَا امْتَحَنَكِ صَابِرَةً وَ زَعَمْنَا أَنَّا لَكِ أَوْلِیَاءُ وَ مُصَدِّقُونَ وَ صَابِرُونَ لِكُلِّ مَا أَتَانَا بِهِ أَبُوكِ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ
  • حضرت زهرا (ع) فرمودند:

    شیعَتُنا مِنْ خِیارِ أهْلِ الْجَنَّهِ وَکُلُّ مُحِبّینا وَ مَوالى اَوْلیائِنا وَ مُعادى أعْدائِنا وَ الْمُسْلِمُ بِقَلْبِهِ وَ لِسانِهِ لَنا.

    شیعیان و پیروان ما، و همچنین دوستداران اولیاء ما و آنان که دشمن دشمنان ما باشند، نیز آن هائى که با قلب و زبان تسلیم ما هستند بهترین افراد بهشتیان خواهند بود.
    وَ أَتَى (أَتَانَا) بِهِ وَصِیُّهُ‏ فَإِنَّا نَسْأَلُكِ إِنْ كُنَّا صَدَّقْنَاكِ إِلاَّ أَلْحَقْتِنَا بِتَصْدِیقِنَا لَهُمَا لِنُبَشِّرَ أَنْفُسَنَا بِأَنَّا قَدْ طَهُرْنَا بِوِلاَیَتِكِ‏ اَلسَّلاَمُ عَلَیْكِ یَا بِنْتَ رَسُولِ اللَّهِ السَّلاَمُ عَلَیْكِ یَا بِنْتَ نَبِیِّ اللَّهِ‏
  • اهمیت دوری از نامحرم

    روزی رسول خدا از اصحاب خود پرسید:« نزدیکترین حالات زن به پروردگارش کدام است؟» اصحاب نتوانستند جواب بدهند. این سؤال به گوش فاطمه علیهاسلام رسید. فاطمه فرمود:

    نزدیکترین حالات زن به پروردگارش وقتی است که در خانه اش بنشیند (و خود را در کوچه و بازار، جلو چشم نا محرمان قرار ندهد.) وقتی رسول خدا این سخن را شنید، فرمود: فاطمه پاره تن من است.
    السَّلاَمُ عَلَیْكِ یَا بِنْتَ حَبِیبِ اللَّهِ السَّلاَمُ عَلَیْكِ یَا بِنْتَ خَلِیلِ اللَّهِ‏ السَّلاَمُ عَلَیْكِ یَا بِنْتَ صَفِیِّ اللَّهِ السَّلاَمُ عَلَیْكِ یَا بِنْتَ أَمِینِ اللَّهِ السَّلاَمُ عَلَیْكِ یَا بِنْتَ خَیْرِ خَلْقِ اللَّهِ‏ السَّلاَمُ عَلَیْكِ یَا بِنْتَ أَفْضَلِ أَنْبِیَاءِ اللَّهِ وَ رُسُلِهِ وَ مَلاَئِكَتِهِ‏
  • عفاف حضرت زهرا (ع)
    زمانی هم که حضرت برای ایراد خطبه وارد مسجد شدنددر عین حفظ حریم و عفت بودند. حضرت امام حسن علیه السلام می فرمایند: مادرم روسری بر سر نمود و چارقد بلند خود را پوشید … در حال راه رفتن بگونه ای بود که لباسش به پا گیر می کرد… او در میان زنان انصار از خانه خارج شد… همچون حضرت رسول صلی الله علیه و آله و سلم با وقار و متانت حرکت می نمود …
    السَّلاَمُ عَلَیْكِ یَا بِنْتَ خَیْرِ الْبَرِیَّةِ السَّلاَمُ عَلَیْكِ یَا سَیِّدَةَ نِسَاءِ الْعَالَمِینَ مِنَ الْأَوَّلِینَ وَ الْآخِرِینَ‏ السَّلاَمُ عَلَیْكِ یَا زَوْجَةَ وَلِیِّ اللَّهِ وَ خَیْرِ الْخَلْقِ بَعْدَ رَسُولِ اللَّهِ‏ السَّلاَمُ عَلَیْكِ یَا أُمَّ الْحَسَنِ وَ الْحُسَیْنِ سَیِّدَیْ شَبَابِ أَهْلِ الْجَنَّةِ
  • قالَتْ (علیها السلام):)

    الرَّجُلُ اُحَقُّ بِصَدْرِ دابَّتِهِ، وَ صَدْرِ فِراشِهِ، وَالصَّلاهِ فى مَنْزِلِهِ إلاَّ الاْمامَ یَجْتَمِعُ النّاسُ عَلَیْهِ.

    هر شخصى نسبت به مرکب سوارى، و فرش منزل خود و برگزارى نماز در آن از دیگرى در أُلویّت است، مگر آن که دیگرى امام جماعت باشد و مردم بخواهند با او نماز جماعت را إقامه نمایند.
    السَّلاَمُ عَلَیْكِ أَیَّتُهَا الصِّدِّیقَةُ الشَّهِیدَةُ السَّلاَمُ عَلَیْكِ أَیَّتُهَا الرَّضِیَّةُ الْمَرْضِیَّةُ السَّلاَمُ عَلَیْكِ أَیَّتُهَا الْفَاضِلَةُ الزَّكِیَّةُ السَّلاَمُ عَلَیْكِ أَیَّتُهَا الْحَوْرَاءُ الْإِنْسِیَّةُ السَّلاَمُ عَلَیْكِ أَیَّتُهَا التَّقِیَّةُ النَّقِیَّةُ السَّلاَمُ عَلَیْكِ أَیَّتُهَا الْمُحَدَّثَةُ الْعَلِیمَةُ
  • قالَتْ (علیها السلام):

    یا أبَا الْحَسَنِ! إنَّ رَسُولَ اللّهِ (صلى الله علیه وآله وسلم) عَهِدَ إلَىَّ وَ حَدَّثَنى أنّى اَوَّلُ أهْلِهِ لُحُوقاً بِهِ وَ لا بُدَّ مِنْهُ، فَاصْبِرْ لاِمْرِاللّهِ تَعالى وَ ارْضَ بِقَضائِهِ.

    اى ابا الحسن! ـ همسرم ـ ، همانا رسول خدا با من عهد بست و اظهار نمود: من اوّل کسى هستم از اهل بیتش که به او ملحق مى شوم و چاره اى از آن نیست، پس تو صبر نما و به قضا و مقدّرات الهى خشنود باش.
    السَّلاَمُ عَلَیْكِ أَیَّتُهَا الْمَظْلُومَةُ الْمَغْصُوبَةُ السَّلاَمُ عَلَیْكِ أَیَّتُهَا الْمُضْطَهَدَةُ الْمَقْهُورَةُ السَّلاَمُ عَلَیْكِ یَا فَاطِمَةُ بِنْتَ رَسُولِ اللَّهِ وَ رَحْمَةُ اللَّهِ وَ بَرَكَاتُهُ صَلَّى اللَّهُ عَلَیْكِ وَ عَلَى رُوحِكِ وَ بَدَنِكِ‏
  • در خصوص حریم نامحرم:

    … إنْ لَمْ یَکُنْ یَرانى فَإنّى أراهُ، وَ هُوَ یَشُمُّ الریح.

    مرد نابینائى وارد منزل شد و حضرت زهراء (علیها السلام)پنهان گشت، وقتى رسول خدا (صلى الله علیه وآله وسلم) علّت آن را جویا شد؟ در پاسخ پدر اظهار داشت: اگر آن نابینا مرا نمى بیند، من او را مى بینم، دیگر آن که مرد، حسّاس است و بوى زن را استشمام مى کند.
    أَشْهَدُ أَنَّكِ مَضَیْتِ عَلَى بَیِّنَةٍ مِنْ رَبِّكِ وَأَنَّ مَنْ سَرَّكِ فَقَدْ سَرَّرَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ‏ وَ مَنْ جَفَاكِ فَقَدْ جَفَا رَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ‏ وَ مَنْ آذَاكِ فَقَدْ آذَى رَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ‏
  • در خصوص تلاوت قرآن:

    فَأکْثِرْ مِنْ تِلاوَهِ الْقُرآنِ، وَالدُّعاءِ، فَإنَّها ساعَهٌ یَحْتاجُ الْمَیِّتُ فیها إلى أُنْسِ الاْحْیاءِ.

    ضمن وصیّتى به امام علىّ (علیه السلام) اظهار نمود: پس از آن که مرا دفن کردى، برایم قرآن را بسیار تلاوت نما، و برایم دعا کن، چون که میّت در چنان موقعیّتى بیش از هر چیز نیازمند به اُنس با زندگان مى باشد.
    وَ مَنْ وَصَلَكِ فَقَدْ وَصَلَ رَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ‏ وَ مَنْ قَطَعَكِ فَقَدْ قَطَعَ رَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ‏ لِأَنَّكِ بَضْعَةٌ مِنْهُ وَ رُوحُهُ الَّذِی بَیْنَ جَنْبَیْهِ كَمَا قَالَ صَلَّى اللَّهُ عَلَیْهِ وَ آلِهِ‏
  • در مورد حجاب فرمودند:

    وَ النَّهْىَ عَنْ شُرْبِ الْخَمْرِ تَنْزیهاً عَنِ الرِّجْسِ، وَاجْتِنابَ الْقَذْفِ حِجاباً عَنِ اللَّعْنَهِ، وَ تَرْکَ السِّرْقَهِ ایجاباً لِلْعِّفَهِ.

    خداوند متعال منع و نهى از شرابخوارى را جهت پاکى جامعه از زشتى ها و جنایت ها; و دورى از تهمت ها و نسبت هاى ناروا را مانع از غضب و نفرین قرار داد; و دزدى نکردن، موجب پاکى جامعه و پاکدامنى افراد مى گردد.
    أُشْهِدُ اللَّهَ وَ رُسُلَهُ وَ مَلاَئِكَتَهُ أَنِّی رَاضٍ عَمَّنْ رَضِیتِ عَنْهُ سَاخِطٌ عَلَى مَنْ سَخِطْتِ عَلَیْهِ‏ مُتَبَرِّئٌ مِمَّنْ تَبَرَّأْتِ مِنْهُ مُوَالٍ لِمَنْ وَالَیْتِ مُعَادٍ لِمَنْ عَادَیْتِ‏ مُبْغِضٌ لِمَنْ أَبْغَضْتِ مُحِبٌّ لِمَنْ أَحْبَبْتِ وَ كَفَى بِاللَّهِ شَهِیداً وَ حَسِیباً وَ جَازِیاً وَ مُثِیباً

حجاب و عفاف از منظر رهبر معظم انقلاب

بانوان مسلمان باید بدانند که خدا، قرآن و اسلام درباره آنها چه قضاوتى دارد، از آنها چه مى‏خواهد و چه مسؤولیتى را براى آنها مطرح کرده است و اجراى آن را نیز خواستار شوند.


محیط تحصیل و جامعه باید براى دختر و پسر سالم و امن باشد و اسلام براى حفظ حدود اخلاقى و کمک به امنیت زن و مرد، حجاب را براى زن تعیین کرده است. زنان مسلمان با حجاب، امنیت خود و مردان را در جامعه تأمین مى‏کنند. آنجا که حجاب را از زن مسلمان دور مى‏کنند، در درجه اول امنیت از زنان و سپس از مردان و جوانان سلب مى‏شود و اسلام براى ایجاد محیطى سالم و داراى امنیت و براى آن که زن بتواند به مسؤولیتهاى اجتماعى خود عمل کند، حجاب را معین کرده است که این خود یکى از احکام برجسته اسلام محسوب مى‏شود.

اگربیاییم با خودمان فکر کنیم که حجاب داشته باشیم، اما چادر نباشد؛ این فکر غلطى است. نه این‌که من بخواهم بگویم چادر، نوع منحصر است؛ نه، من مى‌گویم چادر بهترین نوع حجاب است؛ یک نشانه‌ى ملى ماست؛ هیچ اشکالى هم ندارد؛ هیچ منافاتى با هیچ نوع تحرکى هم در زن ندارد. اگر واقعاً بناى تحرک و کار اجتماعى و کار سیاسى و کار فکرى باشد، لباس رسمى زن مى‌تواند چادر باشد و - همان‌طور که عرض کردم - چادربهترین نوع حجاب است.

البته مى‌توان محجبه بود و چادر هم نداشت؛ منتها همین‌جا هم بایستى آن مرز را پیدا کرد. بعضیها از چادر فرار مى‌کنند، به خاطر این‌که هجوم تبلیغاتى غرب دامنگیرشان نشود؛ منتها از چادر که فرار مى‌کنند، به آن حجاب واقعى بدون چادر هم رو نمى‌آورند؛ چون آن را هم غرب مورد تهاجم قرار مى‌دهد!

زن رکن اساسى و عنصر اصلى خانواده است و اسلام براى نقش زن در داخل خانواده اهمیت فراوانى قائل است زیرا با پایبندى زن به خانواده و تربیت فرزندان در آغوش پرمهر مادر، نسلها در آن جامعه بالنده و رشید خواهند شد.

مقام معظم رهبری در مورد حجاب و عفاف فرمودند : جایگاه زن و نوع حضور او در جامعه اسلامی، یکی از مسائل مهمی است که حتی در تعیین سرنوشت جامعه نقش ایفا می کند.



به گزارش روابط عمومی استانداری بوشهر ؛ حضرت آیت ا... سیدعلی خامنه ای درباره حجاب و عفاف زنان فرمودند، جایگاه زن و نوع حضور او در جامعه اسلامی، یکی از مسائل مهمی است که حتی در تعیین سرنوشت جامعه نقش ایفا می کند. از همین رو دین  اسلام به عنوان کامل ترین ادیان به این موضوع توجه خاصی داشته و احکام و دستورات روشنی در مورد چگونگی حضور زن در اجتماع و اهمیت حجاب برای وی صادر کرده است.


رهبر معظم انقلاب اظهار داشتند : از دیدگاه اسلام حفظ عفت و پاکدامنی زنان نقش تعیین کننده ای در سالم سازی جامعه اسلامی دارد و حتی حفظ جایگاه و منزلت زن به حفظ حجاب و عفت او برمی گردد. در نقطه مقابل یکی از راههای اساسی دشمن جهت ضربه زدن به اسلام و نظام جمهوری اسلامی، ترویج و حمایت از بی بندوباری و مقابله با حجاب و عفاف  است.


وی تصریح کردند : از روزهای اولیه هجوم استعمارگران اروپایی به ممالک اسلامی این حربه برای ضربه زدن به اسلام مورد استفاده قرار گرفته است. در شرایط کنونی نیز دشمن با انواع روشها اعم از راه اندازی شبکه های مبتذل ماهواره ای، ساختن سی دی های مستهجن و ترویج آنها در سطح جامعه، قاچاق مواد الکلی، تبلیغ و ترویج پارتی های مختلط دختران و پسران، حمایت از جریانات فمنیستی و... قصد دارد تا ضمن مقابله با اعتقادات اسلامی و به ابتذال کشاندن جوانان، جامعه ایران را سرگرم مسائل بی ارزش کرده و آن را از دستیابی به آرمانهای خود باز دارد. مقام معظم رهبری در بسیاری از دیدارهایی که با جوانان داشته اند به نوعی به این تهدید اشاره نموده و در جایی می فرمایند: «جوانان ایران اسلامی بدانند که استکبار برای بیرون راندن آنها از صحنه این مبارزه ای که در آن، فتح و پیروزی حتما نصیب ملتهاست و لاغیر، مشغول پول خرج کردن و طراحی انواع نقشه هاست.


مقام معظم رهبری گفتند : هدف این نقشه ها، شما جوانان مؤمن و انقلابی و شجاعی هستید که ایستاده اید، تا هدفهای انقلاب و امامتان را تا آخرین رقم به تحقق برسانید و هر مانعی را از سر این راه بردارید. هر حرکتی که منجر به این بشود که نسل جوان احساس کند باید از صحنه مقاومت کناره بگیرد، این حرکت، باواسطه یا بی واسطه، مربوط به امریکا و نظام استکباری است؛ این یک معیار کلی است. هر کاری که در کشورهای دیگر کردند، این جا هم می خواهند بکنند: سرگرم کردن جوانان به مسائل جنسی، سرگرم کردن جوانان به مسائل شخصی، آلوده کردن جوانان به مواد مخدر و مسکر، منحرف کردن ذهن جوانان از آرمانهای انقلابی و اهداف اصلی، متوجه کردن آنها به اهداف کوچک و جزیی و صنفی و امثال اینها، متزلزل کردن ایمان نسل جوان به اسلام، متزلزل کردن ایمان نسل جوان به اصول انقلاب اسلامی و جمهوری اسلامی و حتی به امام، و نیز هر چیزی که در راه این مبارزه ی عظیم به کار گرفته می شود و وجودش لازم است، آن را در چشم نسل جوان، بی رنگ و بی اثر و بی فایده و مهمل جلوه دادن.


ایشان تأکید کردند : اگر شما جایی چیزی در این مضمونها دیدید، در کتابی خواندید، در رمان ترجمه شده یا نوشته شده ای مطالعه کردید، در مقاله ترجمه شده ای یا در شعر و یا در فلسفه بافی مشاهده کردید، حتی در تظاهرات به اصطلاح جسورانه و پرخشمی چیزی از این قبیل دیدید، آن را متهم کنید؛ بدانید که تصادفی نیست؛ بدانید که امریکا از شما می ترسد.»  (حدیث ولایت، ج 8)با وجود هشدارهای مکرر معظم له در این باره، اما این حقیقت تلخ وجود دارد که در جامعه کنونی ما از دو سو ارزش حجاب و عفاف مورد حمله قرار گرفته است: از یک طرف مظاهر     بی حجابی و بی عفتی گسترش یافته و از طرف دیگر جریانات غربگرای مدعی حقوق زنان (فمنیست ها) با علم کردن ادعای طرفداری از حقوق زن، سعی می کنند دیدگاه لیبرالی غرب در مورد زنان را بر افکار عمومی جامعه غالب کنند. در حالیکه فمنیسم خود پوششی بر استفاده ابزاری از زنان است و بدون اینکه به جوانب حقیقی شخصیت زن توجه داشته باشد و تنزل دادن مقام زن به بهره کشی جنسی توسط غرب را زیر سئوال ببرد، به دنبال مطرح کردن بیشتر آن در اجتماع آن هم با همان ویژگی مورد نظر لیبرالیسم است. از دیدگاه رهبری «در حقیقت، نهضت دفاع از زنان در غرب، یك حركت دستپاچه، یك حركت بی‏منطق، یك حركت مبتنی بر جهالت، بدون تكیه به سنّتهای الهی و بدون تكیه به فطرت و طینت زن و مرد بود كه در نهایت به ضرر همه تمام شد؛ هم به ضرر زنان، هم به ضرر مردان، و بیشتر به ضرر زنان. این قابل تقلید نیست. این فرهنگی نیست كه در كشور اسلامی كسی به آن نگاه كند و بخواهد از آن چیزی یاد بگیرد؛ این را باید طرد كرد.» (30/7/76) این حرکت به مراتب خطرناکتر از گسترش نمودهای بی حجابی است؛ چرا که به صورت  پایه ای دیدگاه اسلام در مورد چگونگی حاضر شدن زن در اجتماع را به چالش می کشد. از این رو باید علاوه بر تبلیغ دیدگاه اسلام در مورد اهمیت حجاب و عفاف، ادعاهای پوچ جریانات غربگرا را نیز پاسخ داد.


نظریه پرداز بزرگ نظام جمهوری اسلامی، حضرت آیت الله العظمی خامنه ای با شناخت نسبت به اهمیت موضوع، مباحث کارشناسی و عالمانه مفیدی در مورد نگاه اسلام به جایگاه زن و لزوم حفظ حجاب و عفاف، دارند که ضروری است جهت دفاع از دیدگاه اسلام در این باره به بیانات معظم له توجه نمود:الف) نقد نگاه غرب به موضوع زن:یکی از مباحث اساسی مقام معظم رهبری پیرامون جایگاه زن، نقد نگاه غرب و آشکار کردن  ظلم های مدعیان کنونی دفاع از حقوق زن نسبت به زنان است.


وی فرمودند : اسلام نگاه بسیار پیشرفته ای به جایگاه زنان دارد و اتفاقا این غرب است که با نگاه ابزاری به زن و سوء استفاده از آن مقام و موقعیت وی را تنزل داده است:«دنیای استكباری سرشار از جاهلیّت، در اشتباه است كه خیال می‏كند ارزش و اعتبار زن به این است كه خود را در چشم مردان آرایش كند تا چشمهای هرزه به او نگاه كنند و از او تمتّع گیرند و او را تحسین نمایند. بساط آن‏چیزی كه امروز به‏عنوان «آزادی زن» در دنیا و از سوی فرهنگ منحطّ غربی پهن‏شده‏است، برپایه این است كه زن را در معرض دید مرد قرار دهند تا از او تمتّعات جنسی ببرند. مردان از آنها لذّت ببرند و زنها وسیله التذاذ مردان شوند. كسانی كه در دنیای جاهل و غافل و گمراه تمدّن غربی ادّعا می‏كنند طرفدار حقوق بشرند، در حقیقت ستمگران به زن هستند.»بر همین اساس معظم له، غرب را ملزم به پاسخگویی درباره چرایی پایمال کردن حقوق زنان  می دانند و می فرمایند:«این، یكی از آن میدانهایی است كه ما در مقابل دنیای استكباری مدّعی هستیم. من بارها به گویندگان و مبلّغین قضیه زن عرض كرده‏ام: این ما نیستیم كه باید از موضع خودمان دفاع كنیم؛ این فرهنگ منحط غرب است كه باید از خودش دفاع كند. آنچه را كه ما برای زن عرضه می‏كنیم، چیزی است كه هیچ انسان اندیشمند با انصافی نمی‏تواند منكر شود كه «این برای زن خوب است.» ما زن را به عفّت، به عصمت، به حجاب، به عدم اختلاط و آمیزش بی‏حد و مرز میان زن و مرد، به حفظ كرامت انسانی، به آرایش نكردن در مقابل مرد بیگانه - برای آن‏كه چشم او لذّت نبرد - دعوت می‏كنیم. این بد است؟ این كرامت زن مسلمان است. این كرامت زن است.


ایشان اظهارداشتند : آنهایی كه زن را تشویق می‏كنند كه خود را به گونه‏ای آرایش دهد كه مردان كوچه و بازار به او نگاه كنند و غرایز شهوانی خودشان را ارضا كنند، باید از خودشان دفاع كنند كه چرا زن را تا این حد پایین می‏آورند و تذلیل می‏كنند؟! آنها باید جواب بدهند. فرهنگ ما، فرهنگی است كه انسانهای والا و اندیشمند غرب هم آن را می‏پسندند و رفتارشان همین‏طور است. در آن‏جا هم خانمهای عفیف و سنگین و متین و زنهایی كه برای خودشان ارزشی قائلند، حاضرنیستند خودشان را برای ارضای غرایز شهوانی بیگانگان و هرزه‏ چشمها وسیله‏ای قرار دهند. فرهنگ منحط غربی، از این قبیل زیاد دارد.»(25/9/71)


مقام معظم رهبری در مورد  ب)نارسائی های فرهنگ غرب در مورد زن فرمودند:«اساسا نگرش غربی نسبت به زن، یك نگرش مبنی بر عدم برابری و عدم تعادل است. شما به شعارهایی كه در غرب داده می‏شود، نگاه نكنید؛ این شعارها پوچ است و حاكی از واقعیت نیست. فرهنگ غربی را از این شعارها نمی‏شود فهمید. فرهنگ غربی را باید در ادبیّات غربی جستجو كرد. كسانی كه با ادبیّات اروپایی، با شعر اروپایی، با رمان و داستان و نمایشنامه‏های اروپایی آشنا هستند، می‏دانند كه در چشم فرهنگ اروپایی، از دوران قرون وسطی‏ و بعد از آن تا اواسط قرن فعلی، زن موجود درجه دوم بوده است! هر چه بر خلاف این ادّعا كنند، خلاف می‏گویند. شما به نمایشنامه‏های معروف شكسپیر انگلیسی نگاه كنید، ببینید با چه نفسی و با چه زبانی و با چه دیدی در این نمایشنامه‏ ها و سایر ادبیّات غربی به زن نگاه می‏شود! مرد در ادبیات غربی، سرور و ارباب زن و اختیاردار اوست، كه بعضی از نمونه های این فرهنگ و آثار آن، امروز هم باقی است.»


ایشان اظهار داشتند : « پایه تمدن امروز اروپایی، همان فرهنگ رومی است. یعنی امروز آنچه كه بر سر تا پای فرهنگ اروپایی و غربی و به تبع آنها فرهنگ امریكایی و خرده‏ریزهای غرب حاكم است، همان اصول و خطوطی است كه در امپراتوری روم وجود داشته؛ همانهاست كه امروز ملاك و معیار است. آن زمان هم زنان را به بالاترین مقام می‏رساندند، احترامشان می‏كردند و به آرایشها و زیورها تزیینشان می‏كردند؛ اما برای چه؟ برای اشباع یكی از خاكی‏ترین و مادّی‏ترین خصلتهای بشری در مرد! این، چه‏قدر توهین و چه‏قدر تحقیر نسبت به انسان و نسبت به جنس زن است!»(4/10/1370)« امروز هم وقتی زنی با مردی ازدواج می‏كند و به خانه شوهر می‏رود، حتّی نام خانوادگی او عوض می‏شود و نام خانوادگی مرد را بر خود می‏گذارد.


مقام معظم رهبری گفتند : در فرهنگ اروپایی، وقتی زن با همه موجودی و املاك خود ازدواج می‏كرد و به خانه شوهر می‏رفت، نه فقط جسم او در اختیار شوهر قرار می‏گرفت، بلكه تمام اموال و املاك و داراییهای او هم كه از پدر و خانواده‏اش به او رسیده بود، متعلّق به شوهر می‏شد! این حقیقتی است كه غربیها نمی‏توانند آن را انكار كنند.»«نكته مهم این است كه حتّی وقتی در اروپا برای زن حقّ مالكیت معیّن شد - طبق بررسی موشكافانه جامعه‏شناسان خودِ اروپا - به خاطر این بود كه كارخانه‏ها كه تازه بساط فناوری مدرن و صنعت در غرب را گسترده بودند، احتیاج به كارگر داشتند؛ اما كارگر كم بود و احتیاج به كارگرِ زیاد احساس می‏شد. لذا برای این كه زنان را به كارخانجات بكشانند و از نیروی كارشان استفاده كنند - كه البته مزد كمتری هم همیشه به زنان می‏دادند - اعلان شد كه زن دارای حقّ مالكیت است! در اوایل قرن بیستم بود كه اروپاییان حقّ مالكیت را به زن دادند.


ایشان ادامه دادند : این، آن نگرش افراطی و غلط و ظالمانه نسبت به زن در غرب و در اروپاست. یك افراطِ این‏چنین، در مقابل تفریط هم دارد. وقتی نهضتی به نفع زنان در چنان فضایی به وجود می‏آید، طبعاً دچار تفریط هایی از طرف مقابل می‏شود. لذا شما می‏بینید در طول چند ده سال، آن‏چنان فساد و بی‏بند و باری در غرب به وسیله آزادی زنان به راه افتاد و رواج پیدا كرد كه خود متفكّران غربی را دچار وحشت كرد! امروز دلسوزان و مصلحان، انسانهای خردمند و باانگیزه در كشورهای غربی، از آنچه پیش آمده است، متوحّش و ناراحتند و البته نمی‏توانند هم جلوِ آن را بگیرند.»(30/7/76)ج)جایگاه زن در نگاه اسلامی:در مقابل نگاه ناقص و اهانت آمیز غرب به موضوع زن، دیدگاه اسلام نگاهی سازنده به زنان دارد؛ چرا که « اسلام برخلاف آنجوری كه همیشه در نظامهای طاغوتی وجود داشته است كه با چشم دیگری اصلاً به زن نگاه میكردند، به زن شخصیت میدهد. در اسلام وقتی برای مؤمن نمونه میخواهند مشخص بكنند، نمونه‌ی زن معین میكنند. «و ضرب اللَّه مثلا للّذین امنوا امرأت فرعون»؛ این نفر اول، «و مریم ابنت عمران»، نفر دوم؛ نمونه برای للذین آمنوا، دو زن. نمونه برای للذین كفروا هم دو زن: «امرأت نوح و امرأت لوط كانتا تحت عبدین من عبادنا ... فخانتاهما».یعنی زن را نه فقط محور و عبرت و آئینه‌ای برای زنان، بلكه برای كل جامعه قرار میدهد. میشد یك مرد انتخاب بكنند. نه، این در مقابله‌ی با آن نگاه انحرافی و غلطی است كه نسبت به زنان وجود داشته. این نگاه، همیشه هم تحقیرآمیز نبوده؛ اما همیشه غلط بوده.»(28/7/88)


مقام معظم رهبری بعد از ترسیم این دیدگاه کلی اسلام در مورد زنان، به نکات دیگری اشاره   می کنند که به طور مشخص نشان دهنده نگاه پیشرفته ای اسلام به حقوق و جایگاه زنان است. در این نگاه، پیشرفت و تعالی زنان در عرصه اجتماع بدون استفاده ابزاری از آنها و تحت ظلم قرار گرفتن آنها مورد هدف است.


خطوط اصلی بیانات معظم له در این باره در ذیل ارائه می شود:1- جلوگیری از تحت ستم بودن زن: « هدف اسلام در دفاع از حقوق زنان این است كه زن تحت ستم نباشد و مردْ خود را حاكم بر زن نداند. در خانواده، حدود و حقوقی هست. مرد حقوقی دارد، زن هم حقوقی دارد و این حقوق به‏شدّت عادلانه و متوازن ترتیب داده شده است.» (30/7/76)


2- زن و مرد از یک جنس هستند: «شما بینید این آیه‏ی شریفه در مورد زن و مرد - عمدتاً در داخل خانواده - چه می‏فرماید. «و من آیاته ان خلق لكم من انفسكم ازواجا» یعنی از جمله‏ی نشانه‏های قدرت الهی یكی این است كه برای شما انسانها، از نوع خودتان و از جنس خودتان همسرانی قرار داد؛ برای شما مردان، زنان را و برای شما زنان، مردان را. از خودتان هستند؛ «من انفسكم»، جنس جداگانه‏ای نیستند، در دو رتبه جداگانه نیستند؛ همه یك حقیقتند، یك جوهر و یك ذاتند. البته در بعضی از خصوصیات، با هم تفاوتهایی دارند؛ چون وظایفشان دو تاست.» (30/7/76)


3- جلوگیری از پایمال شدن حقوق زنان: «طبیعت الهی زن و مرد در محیط خانواده، طبیعتی است كه برای زن و برای مرد رابطه‏ای فیمابین به وجود می‏آورد كه آن رابطه، رابطه عشق و مهربانی است: «مودّة و رحمة». اگر این رابطه تغییر پیدا كرد، اگر مرد در خانه احساس مالكیت كرد، اگر به چشم استخدام و استثمار به زن نگاه كرد، این ظلم است. اگر زن محیط امن و توأم با امنیتی برای درس خواندن، برای كار كردن، برای احیاناً كسب درآمد و برای استراحت نداشته باشد، این ظلم و ستمكاری است. هر كس كه موجب این ظلم شود، باید قانون و جامعه اسلامی با او مقابله كند.» (30/7/76)«چنانچه نگذارند زن به تحصیلات صحیح و به علم و معرفت دست پیدا كند، این ظلم است. اگر شرایط طوری باشد كه زن فرصت پیدا نكند به خاطر زیادی كار و فشار كارهای گوناگون، به اخلاق خود، به دین خود و به معرفت خود برسد، این ظلم است. اگر زن امكان پیدا نكند كه از مایملك خود به طور مستقل و با اراده خود استفاده كند، این ظلم است. اگر به زن در هنگام ازدواج، شوهری تحمیل شود - یعنی خود او در انتخاب شوهر نقشی نداشته باشد و اراده و میل او مورد اعتنا قرار نگیرد - این ظلم است. اگر زن، چه آن وقتی كه داخل خانه با خانواده زندگی می‏كند و چه در صورتی كه از شوهر جدا می‏شود، نتواند از فرزند خود بهره عاطفی لازم را ببرد، این ظلم است. اگر زن استعدادی دارد - مثلاً استعداد علمی دارد، استعداد برای اختراعات و اكتشافات دارد، استعداد سیاسی دارد، استعداد كارهای اجتماعی دارد - اما نمی‏گذراند از این استعداد استفاده كند و این استعداد شكوفا شود، ظلم است.» (30/7/76)


4- دفاع اخلاقی از زن در خانواده: «یك نكته دیگر كه باید مورد توجّه باشد، دفاع اخلاقی و قانونی از زن، بخصوص در داخل خانواده است. دفاع قانونی با اصلاح قوانین و ایجاد قوانین لازم و دفاع اخلاقی با طرح این مسأله و مواجهه با كسانی كه این را درست نمی‏فهمند و زن را در خانه، مستخدمِ مورد ستمِ مرد، بدون تمكّن از خودسازی معنوی تلقّی و این‏گونه هم عمل می‏كنند. باید با این فكر با قوّت و شدت مقابله شود؛ البته به شكل كاملا منطقی و عقلانی.» (30/7/76)معظم له در پایان این بحث هشدار می دهند که در زمینه دفاع از حقوق و جایگاه زنان « از بحثهای انحرافی در این زمینه‏ها باید اجتناب شود. بعضی كسان به عنوان این كه از زن دفاع می‏كنند، به بحثهای انحرافی كشانده می‏شوند و بعضی از مسائل از قبیل دیه و امثال اینها را مطرح می‏كنند؛ در حالی كه اینها بحثهای انحرافی است.» (30/7/76)


د) اهمیت حجاب و عفاف در حفظ حقوق و جایگاه زنان: اما در راستای دفاع از شخصیت حقیقی زن، رعایت حجاب و عفاف مهم ترین مسئله محسوب می شود؛ چرا که از دیدگاه مقام معظم رهبری عامل اصلی پیشرفت مادی و تعالی معنوی زنان حفظ حجاب و عفت است.


ایشان می فرمایند: «ما كه روی حجاب این‏قدر مقیدیم، به خاطر این است كه حفظ حجاب به زن كمك می‏كند تا بتواند به آن رتبه معنوی عالی خود برسد و دچار آن لغزشگاههای بسیار لغزنده‏یی كه سر راهش قرار داده‏اند، نشود.»بنابراین به بیان ایشان « هر حركتی كه برای دفاع از زنان انجام می‏گیرد، باید ركن اصلی آن رعایت عفاف زن باشد. در غرب به خاطر این‏كه به این نكته توجّه نشد - یعنی مسأله عفت زنان مورد اهتمام قرار نگرفت و به آن اعتنایی نكردند - كار به این بی‏بندو باریها كشید. نباید بگذارند عفّت زن - كه مهمترین عنصر برای شخصیت زن است - مورد بی‏اعتنایی قرار گیرد. عفّت در زن، وسیله‏ای برای تعالی و تكریم شخصیت زن در چشم دیگران، حتّی در چشم خودِ مردان شهوتران و بی‏بندوبار است.


رهبر معظم انقلاب اظهار داشتند : عفّت زن، مایه احترام و شخصیت اوست. این مسأله حجاب و مَحرم و نامَحرم و نگاه كردن و نگاه نكردن، همه به خاطر این است كه قضیه عفاف در این بین سالم نگهداشته شود. اسلام به مسأله عفاف زن اهمیت می‏دهد. البته عفاف مرد هم مهمّ است. عفاف مخصوص زنان نیست؛ مردان هم باید عفیف باشند. منتها چون در جامعه، مرد به خاطر قدرت جسمانی و برتری جسمانی، می‏تواند به زن ظلم كند و برخلاف تمایل زن رفتار نماید، روی عفّت زن بیشتر تكیه و احتیاط شده است.» (30/7/76)


وی در مورد لزوم رعایت حجاب در محیط دانشگاهی فرمودند:« محیط دانشگاهها بسیار مهمّ است. در محیط دانشگاهها، خانمهای دانشجو و استاد باید بكوشند روحیه و فرهنگ اسلامی را ترویج كنند. اجازه ندهید به كسانی - اگر خدای نكرده در دانشگاههای كشور هستند - كه نسبت به حجاب اسلامی یا زنان و دانشجویان دخترِ مسلمان بی‏احترامی كنند. اجازه ندهید كه اینها بتوانند افكار فاسد را منتشر كنند. محیط دانشگاه باید محیط اسلامی باشد؛ محیطی برای رشد انسان طراز اسلام؛ انسانی كه الگویش فاطمه زهرا سلام‏اللَّه‏علیهاست. و این، برای آینده كشور فوق‏العاده مهم است.» (25/9/71)


ایشان درباره پرهیز از تحت الشعاع قرار گرفتن عفت و حجاب به خاطر کار کردن عنوان داشتند :«اسلام با كاركردن زن موافق است. نه فقط موافق است، بلكه كار را تا آن‏جا كه مزاحم با شغل اساسی و مهمترین شغل او، یعنی تربیت فرزند و حفظ خانواده نباشد، شاید لازم هم می‏داند. اما این كار نباید با كرامت و ارزش معنوی و انسانی زن منافات داشته‏باشد. نباید زن را تذلیل كنند و او را وادار به تواضع و خضوع نمایند. تكبر از همه انسانها مذموم است، مگر از زنان در مقابل مردان نامحرم! زن باید در مقابل مرد نامحرم متكبّر باشد. «فلا تخضعن بالقول»؛ در حرف‏زدن مقابل مرد نباید حالت خضوع داشته‏باشد. این، برای حفظ كرامت زن است. اسلام این را می‏خواهد و این الگوی زن مسلمان است.» (25/9/71)


وی فرمودند :  بحث پوشش زن نباید از هجوم تبلیغاتی غرب متأثر باشد:« باید توجه كنید كه هیچ بحثی در این زمینه‏های مربوط به پوشش زن، از هجوم تبلیغاتی غرب متأثر نباشد؛ اگر متأثر از آن شد، خراب خواهد شد. مثلاً بیاییم با خودمان فكر كنیم كه حجاب داشته باشیم، اما چادر نباشد؛ این فكر غلطی است. نه این‏كه من بخواهم بگویم چادر، نوع منحصر است؛ نه، من می‏گویم چادر بهترین نوع حجاب است؛ یك نشانه ملی ماست؛ هیچ اشكالی هم ندارد؛ هیچ منافاتی با هیچ نوع تحركی هم در زن ندارد. اگر واقعاً بنای تحرك و كار اجتماعی و كار سیاسی و كار فكری باشد، لباس رسمی زن می‏تواند چادر باشد و - همان‏طور كه عرض كردم - چادر بهترین نوع حجاب است. البته می‏توان محجبه بود و چادر هم نداشت؛ منتها همین‏جا هم بایستی آن مرز را پیدا كرد. بعضیها از چادر فرار می‏كنند، به خاطر این‏كه هجوم تبلیغاتی غرب دامنگیرشان نشود؛ منتها از چادر كه فرار می‏كنند، به آن حجاب واقعی بدون چادر هم رو نمی‏آورند؛ چون آن را هم غرب مورد تهاجم قرار می‏دهد! شما خیال كرده‏اید كه اگر ما چادر را كنار گذاشتیم، فرضاً آن مقنعه كذایی و آن لباسهای "و لیضربن بخمرهنّ علی جیوبهنّ " و همانهایی را كه در قرآن هست، درست كردیم، دست از سر ما بر می‏دارند؟ نه، آنها [غرب]به این چیزها قانع نیستند؛ آنها می‏خواهند همان فرهنگ منحوس خودشان عیناً این‏جا عمل بشود؛ مثل زمان شاه كه عمل می‏شد.» (25/9/71)ه)


مقام معظم رهبری در مراسم تنفیذ حكم ریاست جمهوری سید محمد خاتمی در سال  1380به لزوم برخورد قاطع قانونی با بدحجابی به صراحت فرمودند: «یك جریان فساد هم فسادهای اخلاقی و رواج منكرات است. با این هم باید مقابله و مبارزه كرد. ما خوب می‏دانیم، این جزو تعالیم اسلام است كه با زبان و تبیین باید مردم را با فضایل اخلاقی آشنا كرد و از منكرات دور نگه‏داشت. این به جای خود درست؛ اما با شیوع منكرات و با تظاهرِ به آن باید مقابله كرد. اسلام مرتكبِ منكر را نصیحت و هدایت می‏كند؛ اما حد هم برای او می‏گذارد. با صِرف زبان و توصیه نمی‏شود كاری كرد. قدرت نظام باید جلوِ سیرِ فحشا و فساد را بگیرد. اجازه ندهید كه هوسهای یك عدّه معدود و یك گروه كوچك و اندك در داخل جامعه، موجب اغوای ذهن و فكر دختر و پسر جوان و مرد و زن مؤمنی شود كه هیچ انگیزه فسادی ندارند. شما باید جلوِ آن‏گونه افراد را بگیرید. همه مسؤولان بخشهای مختلف كشور در این زمینه مسؤولند.


وی تأکید کردند : اجازه ندهید عدّه‏ای با تكیه به نام آزادی - كه واقعاً باید بر عنوان مظلوم آزادی گریست كه چه سوء استفاده‏هایی از این نام می‏شود - منكرات و فحشا و بی‏بندوباری را در جامعه رایج كنند. عكس‏العملِ آن این است كه عدّه‏ای به نظام بدبین شوند؛ مثل اوّل مشروطه. البته تخلّف از قانون و خروج از مدار قانونی برای مقابله با هر چیزی كه به نظر انسان منكر می‏آید - بدون اجازه حكومت - خودش یك جرم است؛ مگر نهی از منكر زبانی، كه بارها گفتیم نهی از منكر زبانی جایز و واجب و وظیفه همه است و در هیچ شرایطی هم ساقط نمی‏شود؛ اما آن‏جایی كه نوبت اجرا و عمل برسد، همه باید طبق قوانین عمل كنند. هیچ چیزی مجوّز این نیست كه بگویند چون نیروی انتظامی و قوّه قضایّیه عمل نكردند، خودمان وارد میدان شدیم؛ نخیر، آن روزی كه لازم باشد مردم برای حادثه‏ای خودشان وارد عمل شوند، رهبری صریحاً به آنها خواهد گفت.» (11/5/80)


با توجه به بیانات مقام معظم رهبری در مورد جایگاه زن در دیدگاه اسلامی و اهمیت حجاب و عفاف وی، و نیز با توجه به بحث مقابله با بی حجابی که در این روزها مطرح شده است،  می توان چند نکته را مطرح کرد:1- دیدگاه غرب در مورد زنان، مبتنی بر استفاده ابزاری از زن برای ارضاء نامشروع تمایلات جنسی است و شخصیت زن را صرفا به همین موضوع تنزل داده است. جریانات غربگرای مدعی دفاع از حقوق زنان که در کشور ما نیز فعال هستند، در چارچوب همین نگاه لیبرالی فکر می کنند. در حالیکه اسلام با نگاهی همه جانبه به شخصیت زن، به دنبال پیشرفت مادی و تعالی معنوی زن است بدون اینکه تحت ظلم یا بهره کشی غیر اخلاقی قرار گیرد.


2- در نگاه اسلام، حفظ حجاب و عفاف اصل مهمی در حفظ منزلت زن دارد و نقش تعیین کننده ای در سلامت جامعه اسلامی و ارزش های اسلامی دارد. از همین رو در دیدگاه مقام معظم رهبری توجه ویژه ای به ارزش حجاب برای زنان و لزوم حفظ آن شده است.


3- همانگونه كه مقام معظم رهبری به روشنی اشاره می كنند باید متصدیان قانونی با قاطعیت در مقابل ترویج بی حجابی وابتذال بایستند. در این راه نباید تنها به اقدامات فرهنگی دلخوش كرد؛ بلكه بدحجابی و رواج منكرات پیرامون آن پدیده شومی است كه هم باید نمودهای ظاهری آن را از بین برد و هم ریشه های تقویت این پدیده نظیر تفكرات فمنیستی را خنثی كرد و در این راه بهره گیری از اقدامات قانونی مقتدرانه و تبلیغ فرهنگی باید با یكدیگر لحاظ شود. باید دانست كه برخی از بی حجابی ها، نشأت گرفته از سازماندهی و تشكیلات هدایت شده دشمنان نظام است و سبب می شود تا سایر افراد عادی هم به این وادی كشیده شوند. تهیه و توزیع فیلم های مستهجن، قاچاق مواد الكلی، گسترش ماهواره و... همگی با مشاركت مستقیم و غیر مستقیم دشمن رقم می خورد؛ بنابراین در پدیده ای كه نقش فعال دشمن در آن بارز است، اكتفا كردن به اقدامات فرهنگی نه تنها فایده نخواهد داشت؛ بلكه سبب غفلت از سایر اقدامات شده و حاشیه امنی را برای مروجان بی حجابی فراهم می كند. تركیب اقدامات سخت، نیمه سخت و نرم در این كار راهگشاست، ضمن اینكه همه این اقدامات باید توسط نهادهای قانونی انجام شود.
