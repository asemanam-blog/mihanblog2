شعر ارسالی از مرتضی اربابی حکم آبادی


آمدی محو تماشا شده امشب آسمان

با زمین غرق تمنا شده امشب آسمان


کل عالم مژه بر هم نزند تا به حضور

چشم بگشایی و بیداد کند عقل و شعور


آمدی آبی دریای وجودم باشی ...؟

یا که در قبله گه عشق سجودم باشی ؟


آمدی مست کنی چشم و دلم را با هم

یا که یکدست کنی روح و گِلم را با هم


از نگاهت به دلم ، وای اگر راه دهد

آسمانی زخیال تو به هر ماه دهد


روی قلبم بنویسند اگر نام تو را

یا که در میکده پر کرد زمی جام تو را


تا ننوشم قدحی مست وصالت نشوم

زهر نوشیدم اگر محو خیالت نشوم


تا «مسیح»است، نگاهش به در است ، می آیی

با مسیحی ، کنار ش پدر است ، می آیی

شعر ارسالی از هادی ماهیکار

طــوفان شد و قیــامتِ کبری به پا شدست
کشتار شیــعه جلوه گر کـــربــلـــا شدست


انگشـــتر عقــیـق علی سرخی اش ز خون
یا داغ لالــه های یــمــن بــر ملا شدست ؟


آنجا ســلالــهء حسـنی قطــعه قطــعه شد
اینــجا ســپــاه خیــبـــر حیــدر بنــا شدست


از آسمان حدیــث حمـاســی شندیده شد
قــالَ الرســولُ : نحنُ یمـــانی شنیده شد


بــانــگ فـغــان ویــس قــرن از یــمــان رسید
آنــگه که شمر و خولی و زید و سنان رسید


آل ســعـــود و آل یــــهـــود و ثـــمـــود هــم
با تـیـــغ و خــنجـــر و قدح و خـیـــزران رسید


آیـــات روی نـــیــزه به روشـــنـــگـــری فزود
نــاگــه نــوای مـحتشم روضـه خوان رسید :



باز این چه شورش است فتاده به جانمان
آه از فــغــان شیــعــه کشــی در یمانمان



"تبَّــــت یدیٰ" ســعــودی ملعـــون مزن دگر
پــر شــد ز خـــون آل تشـــیـع یـــمـــن دگر


مــمـتــد مــزن سپــاه عـلــی را صـــدا مزن
جــانش به لب رسیــده امــام حــســن دگر


کــاری مــکــن که امت سلمــان شروع کند
جــنــگ جــهــانــی و شــرر تـن به تـن دگر


اهـــل یمـــن به جـــان محــمـد یمانی ام
ســربــاز صــفــر سیدعلی و حماسی ام


صــبــری بکن غــروب مــحـمد به سر رسد
با بانــگ "یاعلی" به جــهــانی خــبـر رسد


پنهانِ پشــت پردهء پیدا طلوع کند
با لشــکــرش ســتاره و شمس و قمر رسد


با ذوالــفــقــار حیـــدر و انگشتر حسین ...
یعنی که صاحب جبروت و گوهر رسد



مــهــدی ! ببین به حبل متین چنگ میزنیم
ســربـــاز خــیــبــریم و دم جــنــگ میزنیم


مــا در ولـــایـــت عــلــی و عــاشــق قیام
حیــفا ، ریـاض ، در کف ما "ختم این کلام"


جـــان را فـــدای نـــائـــب بر حقـــتان کنیم
از راه دور، بـــر حــــرم مـــادرش ســـلام ...


هــشــدار میــدهیم "جوانک" به هر طریق
زورت نــمــیــرســد به یــمـن ، "زادهء حرام"


آمــاده ایم و منتــظر حــکــم رهــبــریــم
ما ســر سپــردگان عنـایـات کــوثــریــم

چشمت بهشت،من آدم،گندم گنـــــــاه من

شیطان هوای نفس من است و خطای من

کردم گناه،رانده شدم از بهشت چشمـــانت

من را ببخش،که تویی پشت و پنــــاه من

گم کرده ره منم آقا!ز خانــــه ی عشاق

منزل خدا،منـــم عازم،آقا تـــــــو راه من

ای شمع انجمن ارای شام تار و سیــــاه

دنیا تمام نیمه شب آقا،اما تو ماه مــــــن

یک گوشه از نگاه تو باشد دلیل بر اینکه

روزش شود سیـــــاه،اندوه و آه مــــــن

من میکنم دعا که بیایی ز پرده ی غیبت

آمـــــــین بگو تو طبیبا،بر هر دعای من

پریشانی (تقدیم به ساحت مقدس امام زمان (عج)

دیریست که در بند پریشانی خویشم

سرگشته از این بی سرو سامانی خویشم

با هجر سفر کرده ی تنهای زمانه

در کلبه ی سر در گم ویرانی خویشم

"بشکسته تر از خویش ندیدم به همه عمر

افسرده دل از خویشم و زندانی خویشم"1

دریا شده آرام پس از جوشش انبوه

من در هوس کشتی طوفانی خویشم

عمری به گنه بوده ام آلوده ولیکن

امروز دگر غرق پشیمانی خویشم

تا دیده شود شمس فروزان عدالت

در صبح گه قصه بارانی خویشم

گویا همه جا پر شده از وصف جمالش

من شیفته ی سرور نورانی خویشم

تا جلوه نمایی به نظر غایب پیدا

من در پی آیین مسلمانی خویشم

آرام شبی در نظرم جلوه نمودی

انگار هنوز از پی حیرانی خویشم

گفتی که به پایان برسد غصه هجران

افسوس ولی در غم پنهانی خویشم

امید وصال تو دگر تاب مرا برد

من خسته از این حالت بحرانی خویشم

سرور و سالارِ ما مولا علی
عالِمُ الاسرارِ ما مولا علی

ها ولیُّ اللّه ؛ امامُ العالمی
حمدُ للّه دردِ ما را مَرهَمی

ای دوای دردِ ما ؛ رأسُ الکرم
دل هوای وصل دارد در حرم

در حرم دل را مداوا کرده ای
گِــــره های کورِ ما وا کرده ای

در حرم حالم الهی کرده ای
مرده ای را که وَه که ماهی کرده ای

در حرم دارد دلم وصلی مدام
کی رسد احوالِ ما را در کلام

در حرم دادی سُرورم سَرورا
مهرِ مَه دارد عَلَم در ماوَرا

مدح مولا کرده ام گَه در سحر
روحِ ما را کرده دارا دادگر

مدح او هر راه را هموار کرد
مدحِ او در راهِ ما صد کار کرد

ای که عمری مدح او کردی مرور
مدحِ مولا کرده موسی کوهِ طور

مدحِ مولا کرده آدم سوی ما
مدحِ مولا کرده هر گُلروی ما

مدح مولا را محمـّد در سحر
کرده هر دم در دو عالم دادگر

مدحِ مولا کار اللّه ، کار هو
مدعی رَه آورد در موی او

مدحِ مولا کار اللّهُ الصَّمد
مدح او کی حصر گردد در عدد

عالمُ الاسرار را مداحِ او
مسهلُ الاوطار را مداحِ او

هم رسول الله را مداحِ او
هم عدو الله را مداحِ او

هر دلی گم کرده ره در موی او
هر گدا هر دم رود در سوی او

هر که دارد حاصلی در کوی او
گردد هر دم اهلِ هر سرّی مـگو

سائلم مولا . . . مرا حالی عطا
مَـــــر رسد مأوایِ مهدی را گدا

«شعر و شاعری یک نعمت خدادادی است که به صورت جوششی یا کوششی می توان از آن استفاده برد. آنچه که در این میان از اهمیت برخوردار است، به جا آوردن شکر این نعمت و اسراف نکردن در استفاده از این قریحه ی خدادادی است.
شعرا باید با استفاده از محتواهای دینی- مذهبی و مضامین عالی و معارف قوی در شعر، شکر این نعمت را به جا آورند و سطح ذوق و تأمل مردم را بالا برند.»
ما شعرای به اصطلاح مذهبی باید هر چه بیشتر برای تسریع در پیش بینی مقام معظم رهبری مبنی بر نزدیکی یک اوج شعری مجدد در کشور، به تکمیل، پخته تر و عمیق تر کردن شعر های خود بپردازیم.
این امر، جز با دوری از محیط های فرهنگی- هنری فاسد و ایجاد نشست های شعری بین شاعران مذهبی متعهد به آرمان های انقلاب امکان پذیر نمی باشد.
ما نیز سعی کرده ایم تا با ایجاد محیطی مجازی و دعوت از دوستاران شعر آئینی، باشگاهی مجازی به منظور رشد شاعران و شعر آئینی تشکیل دهیم.
به امید موفقیت، و تحقق هر چه زودتر آرمان های انقلاب و رهبری
