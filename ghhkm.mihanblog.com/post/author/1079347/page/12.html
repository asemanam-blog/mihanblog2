    آقا خودم خـوب مـــیدونم كه پسـتم/یه عمـــــره كه دل تو رو شـكستم

    آقا خودم خـوب مــیدونم ســـــیاهم/غـــرق گــــناه و غـــــرق اشتباهم

    غرق ســـــیاهی شـدم و غــرق آه/ولی به جـــــون مادرت كـــــن نگاه

    وقتی مـــــــحرم می رسه خـــمارم/مــــــثال بارون بــــــــــهار می بارم

    وقتی كه روضه می رسه به گوشم/برات آقا پیرهـن ســیاه می پوشم

    منم همون ســــینه زنی كه مسـته/چشمشو از دور و زمــــونه بســـته

    منم همــون ســـــــینه زنی كه زاره/پناهی جــز دســــــــــتای تو نداره

    ســــــــیـنه زنی كه تا ابـــــد یــارته/نوكــــــــــر و خــــادم عـــــــلمدارته

    همــونی كه یه عمــــــریه خـــــرابه/مــــــــــــنتظر دادن یه جــــــــــوابه

    همــــــــونی كه یه عمــــریه باهاته/مــــــــــــنتظر دیدن كـــــــــــربـلاته

    آقا بگــــــــو تـــــو رو خــــــــریدم بیـا/آقا بگــــــــــو بیا توی روضـــــــه ها

    بگو به مــــــــن كه تا ابـــد نوكــــری/زیـــر لوای حضـــــــــــــرت حــیدری

    بگو به من فاطمه (س) اشكاتو دید/تو رو بــــــرای نوكـــــــــــری بـرگزید

    بگو كه امــــــــضا شـــده پای نامت/بیا به كــــــــــربلا ، سرت سلامت !
